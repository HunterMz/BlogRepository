///  冒泡排序
    ///
    ///  - parameter array: 要排序的数组
    ///
    ///  - returns: 排序后的数组
    func BubbleSort(array: [Int]) -> [Int] {
        var count = 0
        var source = array
        for i in 0..<source.count {
            for j in 0..<source.count - i - 1 {
                count += 1
                if source[j] > source[j + 1] {
                    let temp = source[j]
                    source[j] = source[j + 1]
                    source[j + 1] = temp
                }
                
            }
        }
        return source
    }
    
    ///  选择排序
    ///
    ///  - parameter array: 要排序的数组
    ///
    ///  - returns: 排序后的数组
    func SelectionSort(array: [Int]) -> [Int] {
        
        var source = array
        
        for i in 0..<source.count{
            var min = i
            for j in i..<source.count {
                if source[j] < source[min] {
                    min = j
                }
            }
            
            if i != min {
                let temp = source[i]
                source[i] = source[min]
                source[min] = temp
                
            }
        }
        return source
    }
